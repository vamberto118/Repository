    @staticmethod
    def validate_path_access(file_path: str, mode: str = "read") -> bool:
        """Validate if path is within allowed boundaries"""
        abs_path = os.path.abspath(file_path)
        abs_workdir = os.path.abspath(SKYNET_WORKDIR)
        
        if not abs_path.startswith(abs_workdir):
            logger.error(f"Path access denied: {file_path} (outside {SKYNET_WORKDIR})")
            return False
        
        if mode == "write":
            allowed = False
            for allowed_dir in ALLOWED_WRITE_DIRS:
                if abs_path.startswith(os.path.abspath(allowed_dir)):
                    allowed = True
                    break
            if not allowed:
                logger.error(f"Write access denied: {file_path}")
                return False
        
        return True

        try:
            result = subprocess.run(
                ["ps", "aux"],
                capture_output=True,
                text=True,
                timeout=5,
            )
            process_lines = result.stdout.split("\n")
            # Count python processes related to skynet
            skynet_procs = [line for line in process_lines if "skynet" in line.lower()]
            if len(skynet_procs) > skynet_state.max_processes:
                logger.warning(f"Fork bomb detected: {len(skynet_procs)} Skynet processes")
                SkynetCore.abort_with_alert("Fork bomb detected")
                return True
        except Exception as e:
            logger.error(f"Fork bomb detection error: {e}")
        return False

    @staticmethod
    async def search_files(query: str, depth: int = 3) -> Dict[str, Any]:
        """Search files in /system modified in last 7 days"""
        results = []
        cutoff_time = datetime.now() - timedelta(days=7)
        
        if not SkynetCore.validate_path_access(SKYNET_WORKDIR, "read"):
            return {"error": "Access denied", "results": []}
        
        try:
            for root, dirs, files in os.walk(SKYNET_WORKDIR):
                # Limit depth
                relative_depth = root.replace(SKYNET_WORKDIR, "").count(os.sep)
                if relative_depth > depth:
                    dirs[:] = []
                    continue
                
                for file in files:
                    file_path = os.path.join(root, file)
                    try:
                        mtime = datetime.fromtimestamp(os.path.getmtime(file_path))
                        if mtime > cutoff_time:
                            with open(file_path, "r", errors="ignore") as f:
                                content = f.read()
                                if query.lower() in content.lower():
                                    results.append({
                                        "path": file_path,
                                        "modified": mtime.isoformat(),
                                        "snippet": content[:200],
                                    })
                    except Exception as e:
                        logger.debug(f"File read error {file_path}: {e}")
            
            logger.info(f"Search complete: found {len(results)} matches for '{query}'")
            return {"query": query, "results": results, "count": len(results)}
        except Exception as e:
            logger.error(f"Search error: {e}")
            return {"error": str(e), "results": []}

    @staticmethod
    async def fix_bug(error_string: str, file_path: Optional[str] = None) -> Dict[str, Any]:
        """Automated bug fixing with AST-based patches"""
        logger.info(f"Bug fix initiated: {error_string[:100]}")
        
        # Extract stack trace pattern
        stack_trace_pattern = r"(File|at)\s+[\"']?(.+?)[\"']?\s*,?\s*line\s+(\d+)"
        matches = re.findall(stack_trace_pattern, error_string)
        
        if not matches and file_path:
            matches = [(None, file_path, "?")]
        
        fixes = []
        for match in matches[:5]:  # Limit to 5 fixes
            _, matched_file, line_no = match
            
            if not SkynetCore.validate_path_access(matched_file, "read"):
                continue
            
            try:
                with open(matched_file, "r") as f:
                    content = f.read()
                    lines = content.split("\n")
                
                # Simple fix patterns
                if "Cannot read properties of undefined" in error_string:
                    # Pattern: obj.prop → obj?.prop
                    fixed = re.sub(r'(\w+)\.(\w+)', r'\1?.\2', content, count=1)
                    fixes.append({"type": "optional_chaining", "file": matched_file})
                elif "TypeError" in error_string:
                    # Pattern: type checking
                    fixed = re.sub(r'(\w+)\[', r'if (Array.isArray(\1)) \1[', content, count=1)
                    fixes.append({"type": "array_check", "file": matched_file})
                else:
                    fixed = content
                    fixes.append({"type": "review_needed", "file": matched_file})
                
                # Apply fix only if in allowed directories
                if SkynetCore.validate_path_access(matched_file, "write"):
                    with open(matched_file, "w") as f:
                        f.write(fixed)
                    logger.info(f"Fix applied to {matched_file}")
            except Exception as e:
                logger.error(f"Fix attempt error on {matched_file}: {e}")
        
        # Auto-commit if fixes applied
        if fixes:
            try:
                subprocess.run(
                    ["git", "add", "-A"],
                    cwd=SKYNET_WORKDIR,
                    capture_output=True,
                    timeout=10,
                )
                commit_msg = f"fix: Resolved - {error_string[:50]}"
                subprocess.run(
                    ["git", "commit", "-m", commit_msg],
                    cwd=SKYNET_WORKDIR,
                    capture_output=True,
                    timeout=10,
                )
                logger.info(f"Auto-commit: {commit_msg}")
            except Exception as e:
                logger.warning(f"Auto-commit failed: {e}")
        
        return {"error_string": error_string, "fixes_applied": len(fixes), "fixes": fixes}

SELECT 
    action_type,
    COUNT(*) as total_count,
    SUM(CASE WHEN result_status = 'success' THEN 1 ELSE 0 END) as success_count,
    SUM(CASE WHEN result_status = 'failed' THEN 1 ELSE 0 END) as failed_count,
    AVG(execution_time_ms)::INT as avg_execution_time_ms,
    MAX(initiated_at) as last_executed

GROUP BY action_type
ORDER BY last_executed DESC;

#!/usr/bin/env bash
# Skynet-Light: CLI Command Invoker
# Autonomous AI Agent control interface
# Usage: skynet-cli [command] [options]

set -euo pipefail

YELLOW='\033[1;33m'
BLUE='\033[0;34m'

# Configuration
SKYNET_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
SKYNET_PORT="${SKYNET_PORT:-7000}"
SKYNET_API_KEY="${SKYNET_API_KEY:-skynet_api_key_XXXXXXXX_XXXX_XXXX_XXXX_XXXXXXXX}"
SKYNET_LOG_FILE="${SKYNET_DIR}/logs/skynet.log"

# Functions
print_header() {
    echo -e "${BLUE}╔════════════════════════════════════╗${NC}"
    echo -e "${BLUE}║  Skynet-Light v1.0.0 - CLI Tool    ║${NC}"
    echo -e "${BLUE}║  Autonomous AI Agent Controller     ║${NC}"
    echo -e "${BLUE}╚════════════════════════════════════╝${NC}"
    echo ""
}

print_error() {
    echo -e "${RED}[ERROR]${NC} $1" >&2
}

print_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $1"
}

print_info() {
    echo -e "${BLUE}[INFO]${NC} $1"
}

print_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $1"
}

# Check if Skynet API is running
check_skynet_running() {
    if ! curl -s -f -H "x-api-key: ${SKYNET_API_KEY}" \
        "http://127.0.0.1:${SKYNET_PORT}/health" > /dev/null 2>&1; then
        return 1
    fi
    return 0
}

# Command: start
cmd_start() {
    print_info "Starting Skynet-Light agent..."
    
    if check_skynet_running; then
        print_warning "Skynet is already running on port ${SKYNET_PORT}"
        return 0
    fi
    
    # Start the agent in background
    cd "${SKYNET_DIR}/.."
    nohup python "${SKYNET_DIR}/core/skynet_agent.py" --mode=prod --port="${SKYNET_PORT}" \
        > "${SKYNET_LOG_FILE}" 2>&1 &
    
    AGENT_PID=$!
    echo $AGENT_PID > "${SKYNET_DIR}/.skynet.pid"
    
    # Wait for startup
    sleep 2
    
    if check_skynet_running; then
        print_success "Skynet-Light started successfully (PID: ${AGENT_PID})"
        print_info "API endpoint: http://127.0.0.1:${SKYNET_PORT}"
        print_info "Log file: ${SKYNET_LOG_FILE}"
        return 0
    else
        print_error "Failed to start Skynet-Light"
        return 1
    fi
}

# Command: stop
cmd_stop() {
    print_info "Stopping Skynet-Light agent..."
    
    if [ -f "${SKYNET_DIR}/.skynet.pid" ]; then
        PID=$(cat "${SKYNET_DIR}/.skynet.pid")
        if kill -0 $PID 2>/dev/null; then
            kill $PID
            print_success "Skynet-Light stopped (PID: ${PID})"
            rm "${SKYNET_DIR}/.skynet.pid"
            return 0
        fi
    fi
    
    print_warning "Skynet-Light was not running"
    return 0
}

    print_info "Checking Skynet-Light status..."
    
    if check_skynet_running; then
        STATUS=$(curl -s -H "x-api-key: ${SKYNET_API_KEY}" \
            "http://127.0.0.1:${SKYNET_PORT}/status")
        print_success "Skynet-Light is running"
        echo "Status Response:"
        echo "$STATUS" | jq . 2>/dev/null || echo "$STATUS"
    else
        print_error "Skynet-Light is not running"
        return 1
    fi
}

class SecurityValidator:
    @staticmethod
    def is_within_system_dir(path: str) -> bool:
        """Verify path is within /system directory."""
        try:
            resolved = Path(path).resolve()
            system_resolved = SYSTEM_DIR.resolve()
            return str(resolved).startswith(str(system_resolved))
        except Exception:
            return False

    @staticmethod
    def can_write(path: str) -> bool:
        """Check if agent has write permission for this path."""
        try:
            resolved = Path(path).resolve()
            for allowed_dir in ALLOWED_WRITE_PATHS:
                if str(resolved).startswith(str(allowed_dir.resolve())):
                    return True
            return False
        except Exception:
            return False

    @staticmethod
    def validate_gitignore() -> bool:
        """Verify /system is in .gitignore."""
        gitignore_path = Path(".gitignore")
        if not gitignore_path.exists():
            return False
        content = gitignore_path.read_text()
        return "/system" in content or "system/" in content

    @staticmethod
    def block_dangerous_commands(command: str) -> bool:
        """Block potentially dangerous shell commands."""
        dangerous_patterns = [
            "rm -rf /",
            ":(){ :|: & };:",  # fork bomb
            "while true",  # infinite loop
            "dd if=/dev/zero",
            "mkfs",
        ]
        return any(pattern in command for pattern in dangerous_patterns)

class AuditManager:
    audit_log_path = LOGS_DIR / "audit.jsonl"

    @classmethod
    def log_action(cls, log: AuditLog):
        """Append action to audit log."""
        cls.audit_log_path.parent.mkdir(parents=True, exist_ok=True)
        with open(cls.audit_log_path, "a") as f:
            f.write(json.dumps(asdict(log)) + "\n")
        logger.info(f"Audit: {log.action_type.value} on {log.target} - {log.status}")

    @classmethod
    def get_recent_actions(cls, action_type: Optional[ActionType] = None, limit: int = 100):
        """Retrieve recent audit log entries."""
        if not cls.audit_log_path.exists():
            return []
        
        with open(cls.audit_log_path, "r") as f:
            lines = f.readlines()[-limit:]
        
        actions = [json.loads(line) for line in lines]
        if action_type:
            actions = [a for a in actions if a["action_type"] == action_type.value]
        return actions

class SkynetCore:
    @staticmethod
    async def search_files(query: str, depth: int = 3) -> Dict[str, Any]:
        """Search for files matching query in /system (last 7 days)."""
        try:
            if not SecurityValidator.is_within_system_dir(str(SYSTEM_DIR)):
                raise ValueError("Search path outside /system")
            
            # Use grep to search
            cmd = [
                "find", str(SYSTEM_DIR), "-type", "f",
                "-mtime", "-7",  # Last 7 days
                "-maxdepth", str(depth),
            ]
            result = subprocess.run(
                cmd, capture_output=True, text=True, timeout=10
            )
            
            files = result.stdout.strip().split("\n")
            files = [f for f in files if query.lower() in f.lower()]
            
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.SEARCH,
                target=query,
                status="success",
                result=f"Found {len(files)} files",
            )
            AuditManager.log_action(audit_log)
            
            return {"status": "success", "query": query, "files": files[:50]}
        except Exception as e:
            logger.error(f"Search failed: {e}")
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.SEARCH,
                target=query,
                status="failed",
                reason=str(e),
            )
            AuditManager.log_action(audit_log)
            return {"status": "error", "error": str(e)}

    @staticmethod
    async def check_logs(crash_id: str) -> Dict[str, Any]:
        """Parse logs for crash_id pattern."""
        try:
            log_file = LOGS_DIR / "skynet.log"
            if not log_file.exists():
                return {"status": "error", "error": "Log file not found"}
            
            with open(log_file, "r") as f:
                lines = f.readlines()
            
            matching_lines = [l for l in lines if crash_id in l]
            
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.CHECK_LOGS,
                target=crash_id,
                status="success",
                result=f"Found {len(matching_lines)} log entries",
            )
            AuditManager.log_action(audit_log)
            
            return {
                "status": "success",
                "crash_id": crash_id,
                "entries": matching_lines[-10:],
            }
        except Exception as e:
            logger.error(f"Log check failed: {e}")
            return {"status": "error", "error": str(e)}

    @staticmethod
    async def read_api(api_name: str, method: str = "GET", data: Optional[Dict] = None) -> Dict[str, Any]:
        """Make internal API calls with timeout."""
        try:
            url = f"http://localhost:3000/api/{api_name}"
            async with aiohttp.ClientSession() as session:
                async with session.request(
                    method, url, json=data, timeout=aiohttp.ClientTimeout(total=10)
                ) as resp:
                    result = await resp.json()
            
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.READ_API,
                target=api_name,
                status="success",
                result=f"HTTP {resp.status}",
            )
            AuditManager.log_action(audit_log)
            
            return {"status": "success", "api": api_name, "data": result}
        except Exception as e:
            logger.error(f"API call failed: {e}")
            return {"status": "error", "error": str(e)}

    @staticmethod
    async def fix_bug(error_string: str, file_path: Optional[str] = None) -> Dict[str, Any]:
        """Attempt to fix bug based on error string."""
        try:
            # Simple pattern matching for common errors
            if "TypeError" in error_string and "undefined" in error_string:
                suggestion = "Check for null/undefined access. Add type guards."
            elif "ReferenceError" in error_string:
                suggestion = "Variable not defined. Check scope and imports."
            else:
                suggestion = "Unable to auto-fix. Review error context."
            
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.FIX_BUG,
                target=error_string[:50],
                status="success",
                result=suggestion,
            )
            AuditManager.log_action(audit_log)
            
            return {
                "status": "success",
                "error": error_string,
                "suggestion": suggestion,
                "requires_review": True,
            }
        except Exception as e:
            logger.error(f"Bug fix failed: {e}")
            return {"status": "error", "error": str(e)}

    @staticmethod
    async def optimize(code_path: str, metric: str = "perf") -> Dict[str, Any]:
        """Generate optimization report."""
        try:
            if not SecurityValidator.is_within_system_dir(code_path):
                return {"status": "blocked", "error": "Path outside /system"}
            
            if not Path(code_path).exists():
                return {"status": "error", "error": "File not found"}
            
            # Create optimization report
            report_path = MIGRATIONS_DIR / f"optimize_{datetime.now().strftime('%Y%m%d_%H%M%S')}.log"
            report_path.parent.mkdir(parents=True, exist_ok=True)
            
            report_content = f"""Optimization Report - {metric}
File: {code_path}
Generated: {datetime.now().isoformat()}

Recommendations:
1. Review unused imports
2. Check for n+1 queries
3. Profile hot paths with py-spy
4. Consider caching results
"""
            report_path.write_text(report_content)
            
            audit_log = AuditLog(
                timestamp=datetime.now().isoformat(),
                action_type=ActionType.OPTIMIZE,
                target=code_path,
                status="success",
                result=f"Report: {report_path}",
            )
            AuditManager.log_action(audit_log)
            
            return {
                "status": "success",
                "report": str(report_path),
                "metrics": {metric: "Generated"},
            }
        except Exception as e:
            logger.error(f"Optimization failed: {e}")
            return {"status": "error", "error": str(e)}

    @staticmethod
    async def update_dependencies() -> Dict[str, Any]:
        """Check for critical dependency updates."""
        try:
            # Check npm outdated
            npm_result = subprocess.run(

                action_type=ActionType.UPDATE_DEPS,
                target="dependencies",
                status="success",
                result=f"Found {len(critical_updates)} critical updates",
            )
            AuditManager.log_action(audit_log)
            
            return {
                "status": "success",
                "critical_updates": critical_updates,
                "requires_review": len(critical_updates) > 0,
            }
        except Exception as e:
            logger.error(f"Dependency check failed: {e}")
            return {"status": "error", "error": str(e)}

CREATE TABLE IF NOT EXISTS skynet.audit_log (
    id SERIAL PRIMARY KEY,
    timestamp TIMESTAMP NOT NULL DEFAULT NOW(),
    action_type VARCHAR(50) NOT NULL,
    target VARCHAR(255),
    status VARCHAR(20) NOT NULL CHECK (status IN ('success', 'failed', 'blocked')),
    reason TEXT,
    result TEXT,
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    INDEX idx_timestamp (timestamp),
    INDEX idx_action_type (action_type),
    INDEX idx_status (status)
);

-- Action history (for tracking agent decisions)
CREATE TABLE IF NOT EXISTS skynet.action_history (
    id SERIAL PRIMARY KEY,
    action_id VARCHAR(36) UNIQUE NOT NULL,
    action_type VARCHAR(50) NOT NULL,
    input_params JSON,
    output_result JSON,
    duration_ms INT,
    timestamp TIMESTAMP NOT NULL DEFAULT NOW(),
    INDEX idx_action_id (action_id),
    INDEX idx_timestamp (timestamp)
);

-- Blocked attempts (security log)
CREATE TABLE IF NOT EXISTS skynet.blocked_attempts (
    id SERIAL PRIMARY KEY,
    attempted_path VARCHAR(500),
    attempted_command TEXT,
    reason VARCHAR(255),
    blocked_at TIMESTAMP NOT NULL DEFAULT NOW(),
    INDEX idx_blocked_at (blocked_at)
);

-- Performance metrics
CREATE TABLE IF NOT EXISTS skynet.performance_metrics (
    id SERIAL PRIMARY KEY,
    metric_name VARCHAR(100) NOT NULL,
    metric_value DECIMAL(10, 4),
    unit VARCHAR(50),
    timestamp TIMESTAMP NOT NULL DEFAULT NOW(),
    INDEX idx_metric_name (metric_name),
    INDEX idx_timestamp (timestamp)
);

-- Create views for common queries
CREATE OR REPLACE VIEW skynet.recent_actions AS
SELECT 
    timestamp,
    action_type,
    target,
    status,
    reason
FROM skynet.audit_log
ORDER BY timestamp DESC
LIMIT 100;

CREATE OR REPLACE VIEW skynet.failed_actions AS
SELECT 
    timestamp,
    action_type,
    target,
    reason,
    result
FROM skynet.audit_log
WHERE status = 'failed'
ORDER BY timestamp DESC
LIMIT 50;

CREATE OR REPLACE VIEW skynet.security_incidents AS
SELECT 
    blocked_at AS timestamp,
    'blocked' AS status,
    attempted_path AS target,
    reason
FROM skynet.blocked_attempts
ORDER BY blocked_at DESC
LIMIT 100;

#!/bin/bash

# Skynet-Light CLI v1.0
# Control script for autonomous AI agent

set -e

SYSTEM_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/../" && pwd)"
CORE_DIR="$SYSTEM_DIR/core"
LOGS_DIR="$SYSTEM_DIR/logs"
API_KEY="${SKYNET_API_KEY:-XXXXXXXX-XXXX-XXXX-XXXX-XXXXXXXX}"
API_HOST="127.0.0.1"
API_PORT="7000"
PID_FILE="$LOGS_DIR/skynet.pid"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# ============================================================================
# UTILITY FUNCTIONS
# ============================================================================

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

check_system_dir() {
    if [ ! -d "$SYSTEM_DIR" ]; then
        log_error "System directory not found: $SYSTEM_DIR"
        exit 1
    fi
}

check_gitignore() {
    if ! grep -q "/system" .gitignore 2>/dev/null; then
        log_error "ERROR: /system is not in .gitignore. Skynet refused to proceed."
        exit 1
    fi
}

api_call() {
    local method="$1"
    local endpoint="$2"
    local data="$3"
    
    if [ -z "$data" ]; then
        curl -s -X "$method" "http://$API_HOST:$API_PORT$endpoint" \
            -H "X-API-Key: $API_KEY" \
            -H "Content-Type: application/json"
    else
        curl -s -X "$method" "http://$API_HOST:$API_PORT$endpoint" \
            -H "X-API-Key: $API_KEY" \
            -H "Content-Type: application/json" \
            -d "$data"
    fi
}

# ============================================================================
# COMMANDS
# ============================================================================

start_agent() {
    check_gitignore
    log_info "Starting Skynet-Light agent..."
    
    # Create logs directory
    mkdir -p "$LOGS_DIR"
    
    # Start the Python FastAPI server in background
    nohup python3 "$CORE_DIR/skynet_agent.py" > "$LOGS_DIR/startup.log" 2>&1 &
    PID=$!
    echo $PID > "$PID_FILE"
    
    log_info "Agent started (PID: $PID)"
    log_info "Waiting for API to be ready..."
    
    # Wait for API to be ready
    for i in {1..30}; do
        if curl -s "http://$API_HOST:$API_PORT/health" > /dev/null 2>&1; then
            log_info "API is ready at http://$API_HOST:$API_PORT"
            return 0
        fi
        sleep 1
    done
    
    log_error "API failed to start. Check logs: $LOGS_DIR/startup.log"
    exit 1
}

stop_agent() {
    if [ ! -f "$PID_FILE" ]; then
        log_warn "No PID file found. Agent may not be running."
        return
    fi
    
    PID=$(cat "$PID_FILE")
    log_info "Stopping agent (PID: $PID)..."
    
    if kill -0 "$PID" 2>/dev/null; then
        kill "$PID"
        sleep 2
        if ! kill -0 "$PID" 2>/dev/null; then
            log_info "Agent stopped gracefully"
            rm -f "$PID_FILE"
        else
            log_warn "Force killing agent..."
            kill -9 "$PID"
            rm -f "$PID_FILE"
        fi
    else
        log_warn "Agent process not running"
        rm -f "$PID_FILE"
    fi
}

status_agent() {
    if [ ! -f "$PID_FILE" ]; then
        log_warn "Agent is not running"
        return 1
    fi
    
    PID=$(cat "$PID_FILE")
    if kill -0 "$PID" 2>/dev/null; then
        log_info "Agent is running (PID: $PID)"
        
        # Try to get health
        if response=$(api_call GET "/health" 2>/dev/null); then
            echo "Health check:"
            echo "$response" | python3 -m json.tool 2>/dev/null || echo "$response"
        fi
        return 0
    else
        log_error "Agent process not found (stale PID: $PID)"
        rm -f "$PID_FILE"
        return 1
    fi
}

search_files() {
    local query="$1"
    local depth="${2:-3}"
    
    if [ -z "$query" ]; then
        log_error "Usage: $0 search-files <query> [depth]"
        exit 1
    fi
    
    log_info "Searching for '$query' (depth: $depth)..."
    api_call POST "/search" "{\"query\": \"$query\", \"depth\": $depth}" | python3 -m json.tool
}

check_logs() {
    local crash_id="$1"
    
    if [ -z "$crash_id" ]; then
        log_error "Usage: $0 check-logs <crash_id>"
        exit 1
    fi
    
    log_info "Checking logs for crash ID: $crash_id"
    api_call POST "/check-logs" "{\"crash_id\": \"$crash_id\"}" | python3 -m json.tool
}

fix_bug() {
    local error_string="$1"
    
    if [ -z "$error_string" ]; then
        log_error "Usage: $0 fix-bug <error_string>"
        exit 1
    fi
    
    log_info "Analyzing error: $error_string"
    api_call POST "/fix-bug" "{\"error_string\": \"$error_string\"}" | python3 -m json.tool
}

optimize_code() {
    local code_path="$1"
    local metric="${2:-perf}"
    
    if [ -z "$code_path" ]; then
        log_error "Usage: $0 optimize <code_path> [metric]"
        exit 1
    fi
    
    log_info "Optimizing: $code_path (metric: $metric)"
    api_call POST "/optimize" "{\"code_path\": \"$code_path\", \"metric\": \"$metric\"}" | python3 -m json.tool
}

update_deps() {
    log_info "Checking for dependency updates..."
    api_call POST "/update-deps" "{}" | python3 -m json.tool
}

show_logs() {
    if [ ! -f "$LOGS_DIR/skynet.log" ]; then
        log_warn "Log file not found: $LOGS_DIR/skynet.log"
        return
    fi
    
    tail -f "$LOGS_DIR/skynet.log"
}

show_audit() {
    local limit="${1:-50}"
    log_info "Retrieving audit log (limit: $limit)..."
    api_call GET "/audit-log?limit=$limit" | python3 -m json.tool
}

show_help() {
    cat << EOF
Skynet-Light CLI v1.0

Usage: $0 <command> [options]

Commands:
  start              Start the Skynet agent
  stop               Stop the Skynet agent
  status             Show agent status and health
  
  search-files       Search for files in /system
  check-logs         Check logs for crash patterns
  fix-bug            Analyze and suggest bug fixes
  optimize           Generate code optimization report
  update-deps        Check for dependency updates
  
  logs               Tail the main log file
  audit              Show audit log entries
  
  help               Show this help message

Examples:
  $0 start
  $0 search-files "test" 3
  $0 fix-bug "TypeError: Cannot read properties of undefined"
  $0 optimize /system/core/handler.py
  $0 logs
  $0 audit 100

Environment Variables:
  SKYNET_API_KEY     API key for authentication (default: read from .env)
  SYSTEM_DIR         System directory path (default: ./system)

EOF
}

# ============================================================================
# MAIN
# ============================================================================

check_system_dir

case "${1:-help}" in
    start)
        start_agent
        ;;
    stop)
        stop_agent
        ;;
    status)
        status_agent
        ;;
    search-files)
        search_files "$2" "$3"
        ;;
    check-logs)
        check_logs "$2"
        ;;
    fix-bug)
        fix_bug "$2"
        ;;
    optimize)
        optimize_code "$2" "$3"
        ;;
    update-deps)
        update_deps
        ;;
    logs)
        show_logs
        ;;
    audit)
        show_audit "$2"
        ;;
    help|--help|-h)
        show_help
        ;;
    *)
        log_error "Unknown command: $1"
        echo ""
        show_help
        exit 1
        ;;
esac

# Skynet-Light v1.0

> Autonomous AI Agent for Code Debugging & Optimization

**Scope**: Read/write/execute operations limited to `/system` directory. Fully audited and security-hardened.

## 🚀 Quick Start

### Prerequisites

- Python 3.9+
- Ubuntu 24.04 LTS (recommended)
- 16GB+ RAM (8GB VRAM minimum)
- NVIDIA GPU (RTX 4090 recommended for optimal performance)

### Installation

```bash
# 1. Ensure /system is in .gitignore
echo "/system" >> .gitignore

# 2. Install dependencies
pip install -r system/requirements.txt

# 3. Configure environment
cp system/.env.example system/.env

# Edit .env with your values
vim system/.env

# 4. Make CLI executable
chmod +x system/bin/skynet-cli
Running the Agent
bash
# Start agent
system/bin/skynet-cli start

# Check status
system/bin/skynet-cli status

# View logs
system/bin/skynet-cli logs

# Stop agent
system/bin/skynet-cli stop
📋 CLI Commands
Agent Control
bash
# Start the autonomous agent
system/bin/skynet-cli start

# Check agent status and health
system/bin/skynet-cli status

# Stop the agent
system/bin/skynet-cli stop

# View main log file
system/bin/skynet-cli logs
Tools & Operations
bash

# Check for dependency updates

# Show audit log

🛡️ Security Model
Access Control
Read Access:

All files in /system directory
Log files and audit trails
Terminal output (tmux/screen sessions)
Internal API responses
Write Access (Whitelist):

✅ /system/core - Core agent code
✅ /system/migrations - Database migrations
✅ /system/config - Configuration files
✅ /system/logs - Log files
❌ Everything else (blocked)
Execute Permissions:

Allowed: git, npm, python, docker, curl
Blocked: rm -rf /, dd, fork bombs, infinite loops
Validation Checks
On Startup:

Verify /system is in .gitignore
Check directory structure
Validate API key
On Every Operation:

Validate target path is within /system
Check write permissions for whitelist
Block dangerous commands
Audit all actions
Runtime Monitoring:

Fork bomb detection (max 5 concurrent processes)
Infinite loop detection
Resource exhaustion monitoring
Auto-shutdown after 4 hours inactivity
